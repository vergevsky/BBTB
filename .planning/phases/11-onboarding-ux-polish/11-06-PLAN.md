    ```
    import XCTest
    @testable import SettingsFeature
    @testable import Localization

    final class HelpViewTests: XCTestCase {
        func test_allFAQ_keys_resolve_nonEmpty() {
            let pairs = [
                (L10n.helpFaq1Question, L10n.helpFaq1Answer),
                (L10n.helpFaq2Question, L10n.helpFaq2Answer),
                (L10n.helpFaq3Question, L10n.helpFaq3Answer),
                (L10n.helpFaq4Question, L10n.helpFaq4Answer),
                (L10n.helpFaq5Question, L10n.helpFaq5Answer),
            ]
            for (q, a) in pairs {
                XCTAssertFalse(q.isEmpty, "Question is empty")
                XCTAssertFalse(a.isEmpty, "Answer is empty")
                // Защита от unresolved key (резолв вернёт сам key):
                XCTAssertFalse(q.hasPrefix("help.faq"))
                XCTAssertFalse(a.hasPrefix("help.faq"))
            }
        }

        func test_LOC04_FAQ4_contains_detection_keywords() {
            // LOC-04 — FAQ должен явно mention «22 приложения» или «детектирование VPN».
            // Текст может быть на ru или en (тест-runtime locale зависит от CI).
            // Достаточно одного из ключевых слов в любом из языков.
            let q = L10n.helpFaq4Question.lowercased()
            let a = L10n.helpFaq4Answer.lowercased()
            let combined = q + " " + a
            let keywords = ["22", "детект", "detect", "приложен", "applicat", "видят", "see"]
            let foundAny = keywords.contains { combined.contains($0) }
            XCTAssertTrue(foundAny, "FAQ4 текст не содержит маркеров про detection / 22 приложения")
        }

        func test_helpTitle_resolves() {
            XCTAssertFalse(L10n.helpTitle.isEmpty)
            XCTAssertFalse(L10n.helpTitle.hasPrefix("help."))
        }

        func test_helpEntry_resolves() {
            XCTAssertFalse(L10n.helpEntryLabel.isEmpty)
        }
    }
    ```
