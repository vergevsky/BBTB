    **Imports:** `Foundation`, `os`; conditional `import UIKit` (`#if os(iOS)`) / `import AppKit` (`#elseif os(macOS)`).

    **Public surface (cross-platform):**

    1. **protocol URLSchemeQueryable** — minimal abstraction для canOpenURL.
       Один метод: `func canOpenURL(_ url: URL) -> Bool`.
       `Sendable` (для test parallelism).

    2. **protocol WorkspaceQueryable** — minimal abstraction для NSWorkspace.urlForApplication.
       Один метод: `func urlForApplication(withBundleIdentifier identifier: String) -> URL?`.
       `Sendable`.

    3. **public enum MAXDetector** — static-only API:
       - `private static let logger = Logger(subsystem: "app.bbtb.client", category: "detection")`
       - `internal static let iOSSchemeCandidates: [String] = ["max", "max-app", "ru-max", "vkmax"]` — для тестов и для Info.plist sync.
       - `internal static let macOSBundleCandidates: [String] = ["ru.vk.max", "com.vkontakte.max", "chat.max.app", "ru.max.messenger"]`
       - `public static func detectAndLog()` — точка входа; диспатчит по `#if os(...)` к iOS- или macOS-варианту. На обеих платформах внутри вызывает private testable метод:
         - iOS: `internal static func detectIOS(query: URLSchemeQueryable) -> String?` — итерирует `iOSSchemeCandidates`, для каждого строит `URL(string: "\(scheme)://")`, делает `query.canOpenURL(url)`. Возвращает первый matching scheme, иначе nil. Логирует через logger (info, `\(scheme, privacy: .public)` или «not detected, tried N schemes»).
         - macOS: `internal static func detectMacOS(workspace: WorkspaceQueryable) -> String?` — итерирует `macOSBundleCandidates`, для каждого делает `workspace.urlForApplication(withBundleIdentifier: bid)`. Возвращает первый matching bundle id (string), иначе nil. Логирует analogous (info, `bid, privacy: .public` или «not detected»).
       - `public static func detectAndLog()` (real entry) — вызывает testable internal с production query:
         - iOS: `_ = detectIOS(query: RealUIApplication())` где `RealUIApplication: URLSchemeQueryable` обёртка `func canOpenURL(_:) -> Bool { UIApplication.shared.canOpenURL($0) }`.
         - macOS: `_ = detectMacOS(workspace: RealNSWorkspace())` где `RealNSWorkspace: WorkspaceQueryable` обёртка для `NSWorkspace.shared.urlForApplication(...)`.
       - `@MainActor` annotation на real `detectAndLog()` — потому что `UIApplication.shared` requires main actor.

    4. **Real conformances** — приватные struct'ы внутри файла, conditional на `#if os(iOS)` / `#elseif os(macOS)`:
       - `private struct RealUIApplication: URLSchemeQueryable { @MainActor func canOpenURL(_ url: URL) -> Bool { UIApplication.shared.canOpenURL(url) } }` — здесь будет concurrency issue (protocol method not @MainActor, but body требует main). Решение: пометить protocol method `@preconcurrency` или сделать тип `@MainActor`. Простейший вариант — `@MainActor struct RealUIApplication`. Тогда detectIOS вызывать внутри `@MainActor` контекста detectAndLog.
       - `private struct RealNSWorkspace: WorkspaceQueryable { func urlForApplication(...) -> URL? { NSWorkspace.shared.urlForApplication(withBundleIdentifier: identifier) } }`.

    Лог-строки (точные `privacy:` annotations per Pattern S4):
    - При match: `logger.info("MAX-app detected via scheme: \(scheme, privacy: .public)")` (iOS), `logger.info("MAX-app detected via bundle: \(bid, privacy: .public)")` (macOS).
    - При промахе: `logger.info("MAX-app not detected (iOS, tried \(iOSSchemeCandidates.count, privacy: .public) schemes)")` соответственно.
    - НЕ логировать `url.path` с `.public` privacy.

    **CRITICAL constraints:**
    - НЕ публиковать `iOSSchemeCandidates` / `macOSBundleCandidates` как public — только `internal` (тесты в same module через `@testable import`).
    - НЕ хранить detection result в shared mutable state — fire-and-forget per CONTEXT.
    - НЕ trying to query MAX через other means (LSCopyApplicationURLsForBundleIdentifier, etc.) — `canOpenURL` и `urlForApplication` — Apple-canonical.
    - НЕ writing detection result в App Group / Keychain / UserDefaults — D-Detect explicit «только в локальный лог».
    - НЕ delegate detection в Network Extension — это main app only.

    ```
    import XCTest
    @testable import MainScreenFeature

    final class MAXDetectorTests: XCTestCase {
        private final class MockSchemeQuery: URLSchemeQueryable, @unchecked Sendable {
            var registered: Set<String> = []
            func canOpenURL(_ url: URL) -> Bool {
                registered.contains(url.scheme ?? "")
            }
        }

        private final class MockWorkspace: WorkspaceQueryable, @unchecked Sendable {
            var installed: [String: URL] = [:]
            func urlForApplication(withBundleIdentifier identifier: String) -> URL? {
                installed[identifier]
            }
        }

        // iOS path
        #if os(iOS)
        func test_iOS_detectsFirstMatchingScheme() {
            let mock = MockSchemeQuery()
            mock.registered = ["max-app"]
            let result = MAXDetector.detectIOS(query: mock)
            XCTAssertEqual(result, "max-app")
        }
        func test_iOS_returnsNilWhenNoneRegistered() {
            let mock = MockSchemeQuery()
            let result = MAXDetector.detectIOS(query: mock)
            XCTAssertNil(result)
        }
        func test_iOS_prefersFirstCandidate() {
            // Two registered — должен вернуть первый по списку candidates.
            let mock = MockSchemeQuery()
            mock.registered = ["vkmax", "max"]  // оба зареганы
            let result = MAXDetector.detectIOS(query: mock)
            XCTAssertEqual(result, "max")  // первый в iOSSchemeCandidates
        }
        #endif

        // macOS path
        #if os(macOS)
        func test_macOS_detectsFirstMatchingBundle() {
            let mock = MockWorkspace()
            mock.installed["chat.max.app"] = URL(fileURLWithPath: "/Applications/MAX.app")
            let result = MAXDetector.detectMacOS(workspace: mock)
            XCTAssertEqual(result, "chat.max.app")
        }
        func test_macOS_returnsNilWhenNoneInstalled() {
            let mock = MockWorkspace()
            let result = MAXDetector.detectMacOS(workspace: mock)
            XCTAssertNil(result)
        }
        #endif

        // Cross-platform invariants
        func test_candidates_nonEmpty_andNoDuplicates() {
            #if os(iOS)
            XCTAssertFalse(MAXDetector.iOSSchemeCandidates.isEmpty)
            XCTAssertEqual(Set(MAXDetector.iOSSchemeCandidates).count, MAXDetector.iOSSchemeCandidates.count)
            #endif
            #if os(macOS)
            XCTAssertFalse(MAXDetector.macOSBundleCandidates.isEmpty)
            XCTAssertEqual(Set(MAXDetector.macOSBundleCandidates).count, MAXDetector.macOSBundleCandidates.count)
            #endif
        }
    }
    ```
