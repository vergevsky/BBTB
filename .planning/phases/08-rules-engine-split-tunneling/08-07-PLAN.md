```bash
#!/usr/bin/env bash
# Phase 8 / RULES-05 — compile baseline-rules.json → 3 .srs + sign all.
# Invoked manually by developer when baseline-rules.json changes (Pitfall 6 Option A).
# Output: BBTB/Packages/RulesEngine/Sources/RulesEngine/Resources/{bbtb-baseline-*.srs,*.sig,baseline-rules-manifest.json,*.json.sig}
#
# Requirements (per W6 user_setup):
# - sing-box CLI v1.13.x (`brew install sing-box`)
# - openssl 3.x (`brew install openssl@3` — macOS LibreSSL не поддерживает -rawin)
# - jq
# - BBTB_BASELINE_SIGNING_KEY env var → path to Ed25519 private key

set -euo pipefail

REPO_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
RESOURCES="${REPO_ROOT}/Packages/RulesEngine/Sources/RulesEngine/Resources"
BASELINE_JSON="${RESOURCES}/baseline-rules.json"
MANIFEST_OUT="${RESOURCES}/baseline-rules-manifest.json"
MANIFEST_SIG_OUT="${RESOURCES}/baseline-rules-manifest.json.sig"

# Pre-flight checks
command -v sing-box >/dev/null 2>&1 || { echo "ERROR: sing-box CLI not found. brew install sing-box"; exit 1; }
command -v openssl >/dev/null 2>&1 || { echo "ERROR: openssl not found"; exit 1; }
command -v jq >/dev/null 2>&1 || { echo "ERROR: jq not found. brew install jq"; exit 1; }

if [[ -z "${BBTB_BASELINE_SIGNING_KEY:-}" ]]; then
    echo "ERROR: BBTB_BASELINE_SIGNING_KEY env var required (path to Ed25519 private key)"
    exit 1
fi
if [[ ! -f "$BBTB_BASELINE_SIGNING_KEY" ]]; then
    echo "ERROR: BBTB_BASELINE_SIGNING_KEY=$BBTB_BASELINE_SIGNING_KEY does not exist"
    exit 1
fi

# Verify openssl supports Ed25519 sign
if ! echo "test" | openssl pkeyutl -sign -rawin -inkey "$BBTB_BASELINE_SIGNING_KEY" -out /dev/null 2>/dev/null; then
    echo "ERROR: openssl does not support -rawin Ed25519 sign. Use openssl 3.x: brew install openssl@3"
    exit 1
fi

echo "Baseline source: $BASELINE_JSON"
echo "Resources dir:   $RESOURCES"
mkdir -p "$RESOURCES"

# Get sing-box version для manifest
SINGBOX_VERSION=$(sing-box version | head -1 | awk '{print $NF}')
echo "sing-box CLI version: $SINGBOX_VERSION"

# For each category — extract → compile → sign
SHA_BLOCK=""
SHA_NEVER=""
SHA_ALWAYS=""
SIZE_BLOCK=0
SIZE_NEVER=0
SIZE_ALWAYS=0

for category in block never always; do
    case "$category" in
        block)  key="block_completely" ;;
        never)  key="never_through_vpn" ;;
        always) key="always_through_vpn" ;;
    esac

    # Extract category — convert domains/ip_cidrs/countries → headless rule format
    TMP_JSON="/tmp/bbtb-baseline-${category}.json"
    # Headless rule format per sing-box rule-set spec
    jq --arg key "$key" '{
        version: 2,
        rules: [{
            domain: (.[$key].domains // []),
            domain_suffix: [],
            ip_cidr: (.[$key].ip_cidrs // [])
        }]
    }' "$BASELINE_JSON" > "$TMP_JSON"

    SRS_OUT="${RESOURCES}/bbtb-baseline-${category}.srs"
    SIG_OUT="${RESOURCES}/bbtb-baseline-${category}.srs.sig"

    # Compile to .srs binary
    sing-box rule-set compile --output "$SRS_OUT" "$TMP_JSON"
    echo "Compiled: $SRS_OUT ($(wc -c < "$SRS_OUT") bytes)"

    # Sign .srs
    openssl pkeyutl -sign -rawin -inkey "$BBTB_BASELINE_SIGNING_KEY" \
        -in "$SRS_OUT" -out "$SIG_OUT"
    echo "Signed:   $SIG_OUT ($(wc -c < "$SIG_OUT") bytes)"

    # Capture sha256 + size
    SHA=$(shasum -a 256 "$SRS_OUT" | awk '{print $1}')
    SIZE=$(wc -c < "$SRS_OUT" | awk '{print $1}')
    case "$category" in
        block)  SHA_BLOCK="$SHA"; SIZE_BLOCK="$SIZE" ;;
        never)  SHA_NEVER="$SHA"; SIZE_NEVER="$SIZE" ;;
        always) SHA_ALWAYS="$SHA"; SIZE_ALWAYS="$SIZE" ;;
    esac

    rm -f "$TMP_JSON"
done

TOTAL_SIZE=$((SIZE_BLOCK + SIZE_NEVER + SIZE_ALWAYS))

# Build manifest JSON
jq -n \
    --arg version 0 \
    --arg minAppVersion "$(jq -r .min_app_version "$BASELINE_JSON")" \
    --arg shaBlock "$SHA_BLOCK" --argjson sizeBlock "$SIZE_BLOCK" \
    --arg shaNever "$SHA_NEVER" --argjson sizeNever "$SIZE_NEVER" \
    --arg shaAlways "$SHA_ALWAYS" --argjson sizeAlways "$SIZE_ALWAYS" \
    --argjson totalSize "$TOTAL_SIZE" \
    --slurpfile blockCat <(jq '.block_completely' "$BASELINE_JSON") \
    --slurpfile neverCat <(jq '.never_through_vpn' "$BASELINE_JSON") \
    --slurpfile alwaysCat <(jq '.always_through_vpn' "$BASELINE_JSON") \
    '{
        version: ($version | tonumber),
        min_app_version: $minAppVersion,
        srs_format_version: 4,
        total_size_bytes: $totalSize,
        files: [
            {name: "bbtb-baseline-block.srs",  category: "block_completely",   sha256: $shaBlock,  sig_path: "bbtb-baseline-block.srs.sig"},
            {name: "bbtb-baseline-never.srs",  category: "never_through_vpn",  sha256: $shaNever,  sig_path: "bbtb-baseline-never.srs.sig"},
            {name: "bbtb-baseline-always.srs", category: "always_through_vpn", sha256: $shaAlways, sig_path: "bbtb-baseline-always.srs.sig"}
        ],
        block_completely:   $blockCat[0],
        never_through_vpn:  $neverCat[0],
        always_through_vpn: $alwaysCat[0]
    }' > "$MANIFEST_OUT"

echo "Manifest written: $MANIFEST_OUT"

# Sign manifest
openssl pkeyutl -sign -rawin -inkey "$BBTB_BASELINE_SIGNING_KEY" \
    -in "$MANIFEST_OUT" -out "$MANIFEST_SIG_OUT"

echo "Manifest signed: $MANIFEST_SIG_OUT"
echo ""
echo "✓ Build complete. 9 files in $RESOURCES:"
ls -la "$RESOURCES"/{baseline-rules.json,baseline-rules-manifest.json,baseline-rules-manifest.json.sig,bbtb-baseline-*.srs,bbtb-baseline-*.srs.sig}
```
